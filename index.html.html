function doPost(e) {
  try {
    var ss   = SpreadsheetApp.getActiveSpreadsheet();
    var data = JSON.parse(e.postData.contents);
    var tipo = data.tipo || 'factura';
    if (tipo === 'zcierre') {
      var sh = ss.getSheetByName('Cierres Z') || ss.insertSheet('Cierres Z');
      if (sh.getLastRow() === 0) {
        sh.appendRow(['Fecha','Hora','N° Z','Consec',
          'Juguete No','Juguete $','Hogar No','Hogar $',
          'Ferretería No','Ferretería $','Papelería No','Papelería $',
          'Piercings No','Piercings $','Anime No','Anime $',
          'Varios No','Varios $','Otros No','Otros $',
          'Bruto No','Bruto $','Neto No','Neto $',
          'Efectivo','Base Imp','IVA','R.CTA','NS No','GT','Notas']);
        sh.getRange(1,1,1,sh.getLastColumn())
          .setFontWeight('bold').setBackground('#1D9E75').setFontColor('#fff');
      }
      var c = data.cats || {};
      var g = function(k,f){ return (c[k]||{})[f]||0; };
      sh.appendRow([data.fecha,data.hora,data.numero_z,data.consecutivo,
        g('Juguete','no'),g('Juguete','total'),g('Hogar','no'),g('Hogar','total'),
        g('Ferretería','no'),g('Ferretería','total'),g('Papelería','no'),g('Papelería','total'),
        g('Piercings','no'),g('Piercings','total'),g('Anime','no'),g('Anime','total'),
        g('Varios','no'),g('Varios','total'),g('Otros','no'),g('Otros','total'),
        data.bruto_no,data.bruto,data.neto_no,data.neto,
        data.efectivo,data.base_imp,data.iva,data.rcta,data.ns_no,data.gt,data.notas]);
    } else {
      var sh = ss.getSheetByName('Facturas') || ss.insertSheet('Facturas');
      if (sh.getLastRow() === 0) {
        sh.appendRow(['Fecha','Hora','Proveedor','N° Factura','Consecutivo',
          'Categoría','Método Pago','Subtotal','IVA %','IVA $','Total','Ítems','Notas']);
        sh.getRange(1,1,1,sh.getLastColumn())
          .setFontWeight('bold').setBackground('#1D9E75').setFontColor('#fff');
      }
      var items=(data.items||[]).map(function(i){
        return i.descripcion+' x'+i.cantidad+' $'+i.precio_unitario;
      }).join(' | ');
      sh.appendRow([data.fecha,data.hora,data.proveedor,data.numero_factura,
        data.consecutivo,data.categoria,data.metodo_pago,
        data.subtotal,data.iva_rate,data.iva,data.total,items,data.notas]);
    }
    return ContentService.createTextOutput(JSON.stringify({status:'ok'}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status:'error',msg:err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
